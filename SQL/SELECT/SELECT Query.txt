-- FETCH ANNUAL SALARY, ALONG WITH SALARY
-- We can use "" to give aliases with spaces
SELECT
	EMPLOYEE_ID,
	FIRST_NAME||' '||LAST_NAME AS EMPLOYEE_NAME,
	JOB_ID,
	SALARY,
    SALARY * 12 AS "Annual Salary",
	(CASE WHEN COMMISSION_PCT IS NULL THEN 0 ELSE COMMISSION_PCT END * SALARY) + (SALARY * 12) AS "Annual Salary with Commission",
    COMMISSION_PCT
FROM
	HR.EMPLOYEES
